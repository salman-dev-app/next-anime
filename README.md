```bash
git clone https://github.com/salman-dev-app/next-anime.git
cd next-anime
npm install

# Configure Consumet API endpoint
cp .env.example .env.local
# Set NEXT_PUBLIC_API_URL to your Consumet instance

npm run dev
```
